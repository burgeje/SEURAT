Database Upgrade Instructions

If your database is created before 6.0.1, you need to add the following tables before using the UML association features.

 

PATTERNPARTICIPANTS: This table stores the pattern participant information. This includes:

CREATE TABLE PATTERNPARTICIPANTS(
id INTEGER PRIMARY KEY,
pattern_id INTEGER REFERENCES PATTERNS(id) NOT NULL,
name VARCHAR(255) #Name of the participant,
min INTEGER #Minimum number of participants to implement in a class diagram,
max INTEGER #Maximum number of participants to implement in a class diagram,
UNIQUE(pattern_id, name));

 

PART_PART: Participant-Participant relationships

CREATE TABLE PART_PART(
id INTEGER PRIMARY KEY #Unique identification,
part_id INTEGER REFERENCES PATTERNPARTICIPANTS(id) NOT NULL,
ref_id INTEGER REFERENCES PATTERNPARTICIPANTS(id) NOT NULL,
type INTEGER NOT NULL CHECK (type >= 0),
UNIQUE(part_id, ref_id))

 

OPERATIONS: Operations for the pattern participants

CREATE TABLE OPERATIONS(
id INTEGER PRIMARY KEY,
part_id INTEGER REFERENCES PATTERNPARTICIPANTS(id) NOT NULL,
name VARCHAR(255),
UNIQUE(part_id, name));

 

OPERATION_PARTICIPANT: Referenced participants of operation, not including the “is part of” relationship.

CREATE TABLE OPERATION_PARTICIPANT(
id INTEGER PRIMARY KEY,
oper_id INTEGER REFERENCES OPERATIONS(id) NOT NULL,
part_id INTEGER REFERENCES PATTERNPARTICIPANTS(id),
type INTEGER);

 

DIAGRAM_ALTERNATIVE:

CREATE TABLE DIAGRAM_ALTERNATIVE(
id integer primary key,
alt_id integer references alternatives(id) not null,
pattern_id integer references patterns(id) not null,
package_xmi_id varchar(255) not null,
file_path varchar(255) not null);

 

DIAGRAM_PATTERNELEMENTS:

CREATE TABLE DIAGRAM_PATTERNELEMENTS(
id integer primary key,
alt_id integer references alternatives(id) not null,
part_id integer references patternparticipants(id) not null,
xmi_id varchar(255) not null unique,
part2_id integer references patternparticipants(id),
assoc_type integer);
