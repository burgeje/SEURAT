Database Upgrade Instructions

Although no modifications of existing table schema were made in this version, some new tables were added. If your database version is 6.0.0 or older, please add the following table schemas to the database.

 

TACTICS:

CREATE TABLE TACTICS(
id integer NOT NULL,
name varchar(255) NOT NULL UNIQUE,
quality integer NOT NULL REFERENCES ONTENTRIES(ID),
description varchar(255),
time_beh integer DEFAULT 0 check (time_beh >= 0),
PRIMARY KEY (id));

 

TACTIC_PATTERN:

CREATE TABLE TACTIC_PATTERN(
id integer PRIMARY KEY,
tactic_id integer NOT NULL REFERENCES TACTICS(id),
pattern_id integer NOT NULL REFERENCES PATTERNS(id),
struct_change integer NOT NULL CHECK (struct_change >= 0),
num_changes integer NOT NULL CHECK(num_changes>=0),
beh_change integer NOT NULL CHECK(beh_change>=0),
changes integer NOT NULL CHECK(changes >= 0),
description varchar(255),
UNIQUE(tactic_id, pattern_id));

 

TACTIC_NEGONTENTRIES:

CREATE TABLE SEURAT_DUMMY.TACTIC_NEGONTENTRIES(
ID INTEGER PRIMARY KEY,
TACTIC_ID INTEGER NOT NULL REFERENCES SEURAT_DUMMY.TACTICS(ID),
ONT_ID INTEGER NOT NULL REFERENCES SEURAT_DUMMY.ONTENTRIES(ID));
